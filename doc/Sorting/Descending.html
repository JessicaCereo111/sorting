def <=>(other)
  my_value = value

  if @nils && my_value.nil?
    @nils == :nils_first ? -1 : 1
  else
    other_value = other.value

    if @nils && other_value.nil?
      @nils == :nils_first ? 1 : -1
    else
      other_value <=> my_value
    end
  end
end
